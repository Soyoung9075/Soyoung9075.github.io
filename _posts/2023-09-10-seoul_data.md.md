```python
def acq_tax(x) :
    if x <= 60000 :
        return(x * 0.01)
    elif x <= 90000 :
        return(x * ((x/10000) * 2/3 - 3)/100)
    else :
        return(x * 0.03)

# 농어촌특별세
def size_tax(x) :
    if x['공급면적_2']> 85:
        return(x['공급금액'] * 0.002)
    else :
        return(x['공급금액'] * 0)

# 지방교육세
def edu_tax(x) :
    if x <= 60000 :
        return(x * 0.001)
    elif x <= 90000 :
        return((x * ((x/10000) * 2/3 - 3)/100) / 10)
    else :
        return(x * 0.003)
```


```python
# 구매비용_비투기과열지구

def non_spec(x) :
    if x <= 90000 :
        return(x * 0.5 + x * 0.5 * 0.0429 * 2)
    else :
        return(x * 0.7 + x * 0.3 * 0.0429 * 2)

# 구매비용_투기과열지구
def spec(x) :
    if x <= 90000:
        return(x * 0.6 + x * 0.4 * 0.0429 * 2)
    elif x <= 150000:
        return(x * 0.8 + x * 0.2 * 0.0429 * 2)
    else :
        return(x)
```
